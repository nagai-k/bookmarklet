QRコード生成

code 
(function(){
    var img_width=300;
    var img_height=300;
    var message=window.prompt('text', message);
    if(!message){return};
    
    var src_value='http://chart.apis.google.com/chart?chs='+img_width+'x'+img_height+'&cht=qr&chl='+encodeURIComponent(message);
    var img_tag=document.createElement('img');
    img_tag.src=src_value;
    img_tag.width=img_width;
    img_tag.height=img_height;
    img_tag.style['position']='absolute';
    img_tag.style['top']=(screen.height/2 - img_height/2)+'px';
    img_tag.style['left']=(screen.width/2 - img_width/2)+'px';
    img_tag.onclick=function(){
        var delNode=document.getElementById('bmlt_modal');
        delNode.parentNode.removeChild(delNode);
    };
    var div_tag=document.createElement('div');
    div_tag.id='bmlt_modal';
    div_tag.style['width']=screen.width+'px';
    div_tag.style['height']=screen.height+'px';
    div_tag.style['background-color']='black';
    div_tag.style['position']='fixed';
    div_tag.style['z-index']='99999';
    div_tag.style['top']='0';
    div_tag.style['left']='0';
    div_tag.appendChild(img_tag);
    var body_tag=document.getElementsByTagName('body').item(0);
    body_tag.appendChild(div_tag);
})()

ダミー画像作成

code 
(function(){
    if(!message){
        var message = '320x240';
    }
    
    var message=window.prompt('size', message);
    if(!message){return};
    
    var vars = message.split('x');
    if(vars.length !=2){return};
    
    var img_width=parseInt(vars[0], 10);
    var img_height=parseInt(vars[1], 10);
    
    var src_value='http://dummyimage.com/'+message;
    var img_tag=document.createElement('img');
    img_tag.src=src_value;
    img_tag.width=img_width;
    img_tag.height=img_height;
    img_tag.style['position']='absolute';
    img_tag.style['top']=(screen.height/2 - img_height/2)+'px';
    img_tag.style['left']=(screen.width/2 - img_width/2)+'px';
    img_tag.onclick=function(){
        var delNode=document.getElementById('bmlt_modal');
        delNode.parentNode.removeChild(delNode);
    };
    var div_tag=document.createElement('div');
    div_tag.id='bmlt_modal';
    div_tag.style['width']=screen.width+'px';
    div_tag.style['height']=screen.height+'px';
    div_tag.style['background-color']='black';

})()

URLエンコード・デコード

code 
(function(){
    var contents_block=document.createElement('div');
    contents_block.id='contents_block';
    contents_block.style['width']='100%';
    contents_block.style['height']='100%';
    contents_block.style['display']='-webkit-box';
    contents_block.style['-webkit-box-align']='center';
    contents_block.style['-webkit-box-pack']='center';

    var wrapper_block=document.createElement('div');
    wrapper_block.style['width']='90%';
    wrapper_block.style['max-width']='640px';
    wrapper_block.style['text-align']='center';
    
    var encode_description = document.createElement('div');
    encode_description.id='encode_description';
    encode_description.style['width']='100%';
    encode_description.innerHTML='URLエンコード/デコード';
    
    var encode_textarea = document.createElement('textarea');
    encode_textarea.id='encode_textarea';
    encode_textarea.rows = 3;
    encode_textarea.style['width']='98%';
    encode_textarea.style['margin-top']='3%';
    
    var encode_btn = document.createElement('div');
    encode_btn.id='encode_btn';
    encode_btn.innerHTML="エンコード";
    encode_btn.style['color']='white';
    encode_btn.style['background']='black';
    encode_btn.style['padding']='0.5em 0em';
    encode_btn.style['border']='solid white 1px';
    encode_btn.style['border-radius']='2xp';
    encode_btn.style['width']='100%';
    encode_btn.onclick = function(){
        var output_text = encodeURIComponent(document.getElementById('encode_textarea').value);
        var output_block = document.getElementById('decode_textarea');
        output_block.value = output_text;
    };
    
    var decode_textarea = encode_textarea.cloneNode();
    decode_textarea.id='decode_textarea';
    
    var decode_btn = encode_btn.cloneNode();
    decode_btn.innerHTML="デコード";
    decode_btn.onclick = function(){
        var output_text = decodeURIComponent(document.getElementById('decode_textarea').value);
        var output_block = document.getElementById('encode_textarea');
        output_block.value = output_text;
    };
    
    var close_btn = encode_btn.cloneNode();
    close_btn.innerHTML="終了";
    close_btn.onclick = function(){
        var delNode=document.getElementById('bmlt_modal');
        delNode.parentNode.removeChild(delNode);
    };
    close_btn.style['margin-top']='8%';
    close_btn.style['border']='solid gray 1px';
    close_btn.style['color']='gray';
    
    wrapper_block.appendChild(encode_description);
    wrapper_block.appendChild(encode_textarea);
    wrapper_block.appendChild(encode_btn);
    wrapper_block.appendChild(decode_textarea);
    wrapper_block.appendChild(decode_btn);
    wrapper_block.appendChild(close_btn);
    contents_block.appendChild(wrapper_block);
    
    var div_tag=document.createElement('div');
    div_tag.id='bmlt_modal';
    div_tag.style['width']=screen.width+'px';
    div_tag.style['height']=screen.height+'px';
    div_tag.style['background-color']='black';
    div_tag.style['position']='fixed';
    div_tag.style['z-index']='99999';
    div_tag.style['top']='0';
    div_tag.style['left']='0';
    div_tag.style['color']='white';
    div_tag.appendChild(contents_block);
    var body_tag=document.getElementsByTagName('body').item(0);
    body_tag.appendChild(div_tag);
})()
